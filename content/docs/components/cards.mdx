# Cards

Cards are containers that display content and actions about a single subject. They should be easy to scan for relevant and actionable information.

## Basic Card

A basic card with a title and content.

\`\`\`tsx
import { Card, CardContent, CardDescription, CardFooter, CardHeader, CardTitle } from '@/components/ui/card'

export default function BasicCard() {
  return (
    <Card>
      <CardHeader>
        <CardTitle>Card Title</CardTitle>
        <CardDescription>Card Description</CardDescription>
      </CardHeader>
      <CardContent>
        <p>Card Content</p>
      </CardContent>
      <CardFooter>
        <p>Card Footer</p>
      </CardFooter>
    </Card>
  )
}
\`\`\`

## Card with Form

A card containing a form.

\`\`\`tsx
import { Button } from '@/components/ui/button'
import { Card, CardContent, CardDescription, CardFooter, CardHeader, CardTitle } from '@/components/ui/card'
import { Input } from '@/components/ui/input'
import { Label } from '@/components/ui/label'

export default function CardWithForm() {
  return (
    <Card className="w-[350px]">
      <CardHeader>
        <CardTitle>Create project</CardTitle>
        <CardDescription>Deploy your new project in one-click.</CardDescription>
      </CardHeader>
      <CardContent>
        <form>
          <div className="grid w-full items-center gap-4">
            <div className="flex flex-col space-y-1.5">
              <Label htmlFor="name">Name</Label>
              <Input id="name" placeholder="Name of your project" />
            </div>
            <div className="flex flex-col space-y-1.5">
              <Label htmlFor="framework">Framework</Label>
              <Input id="framework" placeholder="Next.js" />
            </div>
          </div>
        </form>
      </CardContent>
      <CardFooter className="flex justify-between">
        <Button variant="outline">Cancel</Button>
        <Button>Deploy</Button>
      </CardFooter>
    </Card>
  )
}
\`\`\`

## Card with Image

A card with an image.

\`\`\`tsx
import { Card, CardContent, CardDescription, CardFooter, CardHeader, CardTitle } from '@/components/ui/card'
import Image from 'next/image'

export default function CardWithImage() {
  return (
    <Card className="w-[350px] overflow-hidden">
      <div className="relative h-[200px]">
        <Image
          src="/placeholder.svg?height=200&width=350"
          alt="Card image"
          fill
          className="object-cover"
        />
      </div>
      <CardHeader>
        <CardTitle>Card Title</CardTitle>
        <CardDescription>Card Description</CardDescription>
      </CardHeader>
      <CardContent>
        <p>Card Content</p>
      </CardContent>
      <CardFooter>
        <p>Card Footer</p>
      </CardFooter>
    </Card>
  )
}
\`\`\`

## Interactive Card

A card with interactive elements.

\`\`\`tsx
import { Button } from '@/components/ui/button'
import { Card, CardContent, CardDescription, CardFooter, CardHeader, CardTitle } from '@/components/ui/card'
import { Badge } from '@/components/ui/badge'
import { Heart } from 'lucide-react'

export default function InteractiveCard() {
  return (
    <Card className="w-[350px] transition-all hover:shadow-lg">
      <CardHeader>
        <div className="flex items-center justify-between">
          <CardTitle>Interactive Card</CardTitle>
          <Badge>New</Badge>
        </div>
        <CardDescription>This card has interactive elements</CardDescription>
      </CardHeader>
      <CardContent>
        <p>Hover over the card to see a shadow effect.</p>
      </CardContent>
      <CardFooter className="flex justify-between">
        <Button variant="outline" size="icon">
          <Heart className="h-4 w-4" />
        </Button>
        <Button>View Details</Button>
      </CardFooter>
    </Card>
  )
}
\`\`\`
